function fizzbuzz(n) {
    let result = [];
    for (let i = 0; i &lt; n; i++) {
        result.push(i + 1);
        if((result[i] % 3 === 0) && (result[i] % 5 === 0)) {
            result[i] = 'FizzBuzz';
        }

            result[i] = 'Fizz';
